```bash
EVID=.planning/phases/06-pre-launch-qa/evidence/build/bundle-secret-grep.txt
{
  echo "# Client-bundle security grep (D-14)"
  echo "Date: $(date -u +%Y-%m-%dT%H:%M:%SZ)"
  echo "Commit SHA: $(git rev-parse HEAD)"
  echo ""
  echo "D-14 requires scanning BOTH build-output trees:"
  echo "  1. .next/static/chunks/ (from next build)"
  echo "  2. .vercel/output/static/_next/static/chunks/ (from vercel build)"
  echo "Plus a broader .vercel/output/ sweep for function bundles and prerendered HTML."
  echo ""

  # Part A: the two chunk roots named explicitly in D-14
  for ROOT in .next/static/chunks .vercel/output/static/_next/static/chunks; do
    echo "## Scanning $ROOT"
    if [ ! -d "$ROOT" ]; then
      echo "  (directory does not exist, skipping — see note below)"
      echo ""
      continue
    fi
    echo "  File count: $(find "$ROOT" -name '*.js' 2>/dev/null | wc -l | tr -d ' ')"

    for PATTERN in "NOTION_TOKEN" "secret_" "ntn_"; do
      echo "  ### grep '$PATTERN':"
      HITS=$(grep -rln "$PATTERN" "$ROOT" 2>/dev/null || true)
      if [ -z "$HITS" ]; then
        echo "    (no matches, OK)"
      else
        echo "$HITS" | sed 's/^/    LEAK: /'
      fi
    done

    if [ -n "$NOTION_DATABASE_ID" ]; then
      echo "  ### grep NOTION_DATABASE_ID literal value:"
      HITS=$(grep -rln "$NOTION_DATABASE_ID" "$ROOT" 2>/dev/null || true)
      if [ -z "$HITS" ]; then
        echo "    (no matches, OK)"
      else
        echo "$HITS" | sed 's/^/    LEAK: /'
      fi
    fi
    echo ""
  done

  # Part B: broader .vercel/output/ sweep (D-14 "for good measure")
  echo "## Broader sweep: .vercel/output/ (all files)"
  if [ ! -d .vercel/output ]; then
    echo "  (directory does not exist — Task 2 vercel build did not populate it, investigate)"

    for PATTERN in "NOTION_TOKEN" "secret_" "ntn_"; do
      echo "  ### grep '$PATTERN' recursively under .vercel/output/:"
      HITS=$(grep -rln "$PATTERN" .vercel/output 2>/dev/null || true)
      if [ -z "$HITS" ]; then
        echo "    (no matches, OK)"

        echo "$HITS" | sed 's/^/    LEAK: /'
      fi
    done

  echo "## Server-only module spot-check"
  if grep -l "'use client'" src/lib/notion.ts src/lib/notion-projects.ts 2>/dev/null; then

} > "$EVID"
```
